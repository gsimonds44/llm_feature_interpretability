    board = [[' ' for _ in range(n)] for _ in range(n)]

        for j in range(n):
            if (i+j) % 2 == 0:
                board[i][j] = 'X'
            else:
                board[i][j] = 'O'
    for row in board:

def pascal_triangle(n):
    triangle = [[1]*(i+1) for i in range(n)]
    for i in range(2, n):
        for j in range(1, i):
            triangle[i][j] = triangle[i-1][j-1] + triangle[i-1][j]
    return triangle
